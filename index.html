        Chosen Palette: "Cosmic Harmony". It uses a base of warm neutrals (stone) for the background to create a calm, grounded feeling.     <!-- 

        Application Structure Plan: The SPA is designed around a central, interactive "Concentric Flywheel" diagram. This structure was chosen over a linear, scrolling report because the source material describes a non-linear, cyclical system. The flywheel model allows users to grasp the entire system bottlene at a glance, then dive into specific layers at their own pace by clicking on them. This user-driven exploration ("overview first, zoom and filter, then details-on-demand") is ideal for understanding complex, interconnected concepts. The flow is intuitive: a click on a visual element reveals corresponding detailed information in a dedicated panel, keeping the user oriented and engaged within a single, unified view.

    -->

    <!-- 

        Visualization & Content Choices: 

        1. Flywheel Diagram: Built with structured HTML (nested divs) and styled with Tailwind CSS. Goal: Create a responsive, accessible, and easily maintainable visualization without relying on complex SVG or Canvas libraries, which are unnecessary for this geometric design. Interaction: Click events handled by vanilla JS to update content dynamically. Justification: This method is lightweight, performant, and aligns perfectly with the single-file HTML requirement.

        2. Icons: Unicode characters (e.g., '→', '⚡️', '⚙️') are used. Goal: Provide simple, universally understood visual cues. Justification: Zero-dependency, fast-loading, and easy to implement.

        3. Animations: Subtle CSS transitions on hover and click. Goal: Enhance user feedback and give the system a "live", responsive feel. Justification: Native CSS animations are highly performant and easy to implement with Tailwind's utility classes.

        4. No Charts: The source material is purely conceptual, containing no quantitative data, so no charts (Chart.js/Plotly.js) are needed.

    -->

        let activeLayer = null;


        rings.forEach(ring => {

            ring.addEventListener('click', (e) => {

                e.stopPropagation(); 

                const layerId = ring.dataset.layer;

                

                if (activeLayer === layerId) {

                    resetToDefault();

                    return;

                }

                

                updateContent(layerId);

                updateActiveRing(ring);

                activeLayer = layerId;

            });

        });

        

        document.getElementById('flywheel-section').addEventListener('click', () => {

             resetToDefault();

        });


        function updateContent(layerId) {

            const data = flywheelData[layerId];

            const targetPanel = contentPanels[layerId];


            Object.values(contentPanels).forEach(panel => {

                panel.classList.add('hidden');

                panel.classList.remove('visible');

            });

            

            targetPanel.innerHTML = `

                <h2 class="text-2xl md:text-3xl font-bold mb-2" style="color: ${data.color};">${data.title}</h2>

                <p class="text-lg text-gray-500 mb-6">${data.subtitle}</p>

                <div class="prose max-w-none prose-p:text-gray-700 prose-strong:text-gray-900">

                   ${data.content}

                </div>

            `;

            targetPanel.classList.remove('hidden');

            setTimeout(() => targetPanel.classList.add('visible'), 50);

        }


        function updateActiveRing(activeRing) {

            rings.forEach(r => {

                r.classList.remove('active');

                r.style.borderWidth = '4px';

            });

            activeRing.classList.add('active');

            activeRing.style.borderWidth = '6px';

        }

        

        function resetToDefault() {

            Object.values(contentPanels).forEach(panel => {

                if (panel.id !== 'content-panel-default') {

                    panel.classList.add('hidden');

                    panel.classList.remove('visible');

                }

            });

            contentPanels.default.classList.remove('hidden');

            setTimeout(() => contentPanels.default.classList.add('visible'), 50);

            

            rings.forEach(r => {

                r.style.borderWidth = '4px';

            });
